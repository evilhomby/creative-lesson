    # 關於 JavaScript 的數值常用操作

    # 四則運算 (加減乘除)

    運算順序：先乘除後加減，由左至右、括號優先

    var a = 10;
var b = 20;
console.log(a + b);
console.log(a - b);
console.log(a * b);
console.log(a / b);

    # 取餘數

    餘數永遠小於除數，在 n 個一組進行輪替、奇偶數的情況下會用到。例：奇偶奇偶奇偶，ABCABCABC (三個一組輪替)

    var a = 5;
var b = 2;
console.log(a % b);
    # 實作練習：將 52 張撲克牌依數字大小排序
    var cardNum = 52;
var suits = ['黑桃', '紅心', '方塊', '梅花'];
for(var i = 0; i < cardNum; i++) {
	var suit = i % 4; // 每 4 個一組 
	var number = Math.floor(i / 4) + 1;
	console.log(suits[suit] + '-' + number);
}

    依花色排序的話怎麼做呢？

    var cardNum = 52;
var suits = ['黑桃', '紅心', '方塊', '梅花'];
for(var i = 0; i < cardNum; i++) {
	var suit = Math.floor(i / 13);
	var number = i % 13 + 1;
	console.log(suits[suit] + '-' + number);
}

    # 四捨五入、無條件進位與捨去 (取整數)

    使用的是 Math 物件的以下三個方法

    • round(): 四捨五入
    • ceil(): 無條件進位
    • floor(): 無條件捨去
    var n = 3.1415;
console.log(Math.round(n));
console.log(Math.ceil(n));
console.log(Math.floor(n));

    取小數點後第 n 位的做法？

    // 取小數點後一位
var num = 3.15;
console.log(Math.round(num * 10) / 10); // 3.2

    # 取最大值、最小值

    var a = 10
var b = 100;
var c = 1000;
console.log(Math.max(a, b, c)); // 1000
console.log(Math.min(a, b, c)); // 10
    # 限制使用者輸入 0-100 之間的數值
    // 假設 input 為使用者輸入的數值
var input = 99;
var result = Math.max(Math.min(input, 100), 0);

    # 隨機數

    JavaScript 使用 Math 物件的 random() 方法來產生隨機數，範圍是 0 ≦ random ＜ 1

    例如：取 0 至 100 間的整數

    // 0 * 101 ≦ random ＜ 1 * 101 之後無條件捨去
Math.floor(Math.random() * (100 + 1));

    例如：取 0 或 1

    Math.round(Math.random());
Math.floor(Math.random() * 2);

    取 0 或 1 可以用 Math.round()，但其他清況不建議，因為最小值與最大值的機率與其他數字不同。

    # 實作練習
    • 請用函式寫出取 1 到 n 數的整數
    var min = 1;
var max = 100;
var random = Math.floor(Math.random() * (max - min + 1)) + min;
    • 隨機取出 1 與 -1 (此例適用於 n 個中取 1 個)
    // 用陣列去操作 (推薦! 直覺且彈性大 )
[1, -1][Math.floor(Math.random() * 2)]

// 1 減去 0 或 2
1 - Math.floor(Math.random() * 2) * 2

// 用條件判斷 (以下兩種寫法結果相同)
Math.random() < 0.5 ? -1 : 1;

var result;
if(Math.random() < 0.5) {
	result = -1;
}else{
	result = 1;
}

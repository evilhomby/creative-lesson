    # JavaScript 基礎 II

    練習時可以使用 Chrome 的開發人員工具 > 主控台 (Mac 快速鍵: cmd + option + i) 來輸入 js 程式碼。

    清除 Console 可按右鍵 Clear Console 或快速鍵 ctrl + L；Clear Console History 則會清除輸入過的程式碼記錄。

    在輸入狀態按「上或下」可以叫出之前執行過的程式

    # 條件判斷

    條件判斷需搭配布林值，依判斷結果為 True 或 False 決定執行動作。可使用 == (一般相等) 或 === (嚴格相等) 及 !=, !==, >, >=, <, <= 這些比較運算子。

    //如果判斷式成立執行花括號內程式
if(true) {
	alert('hello');
}

    若成立執行 A，不成立執行 B

    var age = 17;
if(age >= 20) {
	alert('成年');
}else{
	alert('未成年');
}

    判斷式超過兩個以上，可以使用 if-elseif-else 或 switch 的寫法。(switch 是用嚴格相等進行判斷)

    var age = 38;
if(age >= 40) {
	alert('借卵');
}else if(age >= 38){
	alert('三代試管 Plus');
}else if(age >= 35) {
	alert('試管嬰兒');
}else if(age >= 20) {
	alert('凍卵');
}else{
	alert('小於 20 歲');
}

    var pet = 'cat';
switch (pet) {
	case 'dog':
		alert('you have a dog');
		break;
	case 'cat':
		alert('you have a cat');
		break;
	case 'pig':
		alert('you have a pig');
		break;
	default:
		alert('you do not have a pet');
}

    另外可以使用邏輯運算子來判斷多個條件同時成立、或擇一成立的情況

    //性別為女性且年齡大於等於 20 歲可以捐卵
var gender = 'female';
var age = 25;
if(gender == 'female' && age >= 20) {
	alert('可以捐卵');
}

    //身心障礙或年齡小於等於 10 歲半價收費
var disabled = false;
var age = 6;
if(disabled == true || age <= 10) {
	alert('半價');
}else{
	alert('原價');
}

    上例也可以改用巢狀的判斷達成，但較不直覺。使用上若可以把發生機率較高者放在前面，執行效率會比較高。

    var disabled = false;
var age = 6;
if(disabled == true) {
	alert('半價');
}else{
	if(age <= 10) {
		alert('半價');
	}else{
		alert('原價');
	}
}
    # 實例：按鈕點擊後切換 RGB 三種顏色
    <div id="btn" style="background:red;">example</div>
<script>
var color = 'red';
document.querySelector('#btn').onclick = function() {
	if(color == 'red') {
		color = 'green';
	}else if(color == 'green') {
		color = 'blue';
	}else{
		color = 'red';
	}
	this.style.background = color;
}
</script>

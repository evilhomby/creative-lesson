    # 關於 JavaScript 的字串常用操作

    複習一下字串需要用單引號、雙引號或是反引號包住，通常會使用單引號將雙引號留給 HTML 使用。

    // 單行可用單引號或雙引號
var a = 'abc';
var b = "def";

// 多行需使用反引號
var c = `line1
line2
`;

// 字串為 HTML 的狀況
var html = '<div id="example"></div>';

    # 字串的串接

    直接用 + 運算子就可以囉！(補充：什麼是運算子和運算元呢？)

    當字串與數值混著串接時，需注意運算順序或使用括號，避免產生非預期的結果。

    var a = 'hello';
var b = 'world'
var c = a + ' ' + b;
console.log(c); // hello world

    # 取得字串長度

    JavaScript 中文英文都算是一個字元/長度 (有的程式語言中文不是一個字元)

    var str = '一二三abc'
console.log(str.length);

    # 取得字串的一部分

    使用的是 String 物件的 substr() 與 substring() 方法，這兩個方法差別在於第二個參數。如省略第二個參數即代表截取至字串的最後一個字元。

    • substr(start, length): 從 start 開始取 length 個
    • substring(start, end): 從 start 開始取到 end (個數 = end - start)
    var str = '123456789';
console.log(str.substr(3, 3)); // 456
console.log(str.substring(3, 6)); // 456
console.log(str.substr(3)); // 456789

    # 刪除字串前後的空白

    使用 trim() 方法即可

    var str = ' Hello World ';
str = str.trim();
console.log(str);
    • trimStart() 刪除字串開頭的空白 (左邊)
    • trimEnd() 刪除字串結尾的空白 (右邊)

    # 大小寫的轉換

    • toLowerCase(): 轉成小寫
    • toUpperCase(): 轉成大寫
    var str = 'HeLlO WoRlD';
console.log(str.toUpperCase());
console.log(str.toLowerCase());
    # 實作練習
    • 首字大寫沒有內建的方法，需要自行撰寫轉換程式
    var str = 'hello world';
str.substr(0, 1).toUpperCase() + str.substr(1).toLowerCase();

    實作為一個函式

    function upperCaseFirst(str) {
	return str.substr(0, 1).toUpperCase() + str.substr(1).toLowerCase();
}
console.log(upperCaseFirst('hello world')); // Hello world
console.log(upperCaseFirst('HELLO WORLD')); // Hello world

    # 字串的尋找

    檢查字串中是否有符合的子字串，有的話傳回索引值，沒有的話傳回 -1

    • indexOf(): 第一個相符的
    • lastIndexOf(): 最後一個相符的
    var str = 'hello world, I love world';
console.log(str.indexOf('world')); // 6
console.log(str.lastIndexOf('world')); // 20

if(str.indexOf('world') != -1){
	console.log('有world！');
}

    # 把字串變成陣列

    使用 split() 方法，參數是分隔符號，例如：逗點、空白…等。

    var str = 'a,b,c,d,e';
console.log(str.split(',')); // ['a', 'b', 'c', 'd', 'e']

    # 實作練習

    # 自「中文名．EnglishName」格式的字串分別取出中文名與英文名
    // 取得中文名
// 使用 indexOf() 找到分隔符號的位置即為中文名的字數
function cname(str) {
	var end = str.indexOf('．');
	return str.substr(0, end);
}
console.log(cname('送子鳥．Stork')); 

// 取得英文名
// 使用 indexOf() 找到分隔符號的位置，後移一位以 substr 截取
function ename(str) {
	var start = str.indexOf('．') + 1;
	return str.substr(start);
}
console.log(ename('送子鳥．Stork')); 

    // 使用 split 將字串轉為陣列，並傳入索引 (號碼) 取得需要的資訊
function getInfo(str, i) {
	var arr = str.split('．');
	return arr[i];
}
console.log(getInfo('送子鳥．Stork', 0)); // 送子鳥
console.log(getInfo('送子鳥．Stork', 1)); // Stork
    # 如何將字串前後反轉？例：Hello World 變成 dlroW olleH
    // 使用陣列的 reverse() 將順序反轉後，再以 join() 合併為字串
var str = 'Hello World';
var reverse = str.split('').reverse().join('');
console.log(reverse);

// 使用 substr() 自行連接字串
var str = 'Hello World';
var reverse = '';
for(var i = 1; i <= str.length; i++) {
	reverse += str.substr(str.length - i, 1);
}
console.log(reverse);

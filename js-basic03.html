    # JavaScript 基礎 III

    # 迴圈

    迴圈用在於「重覆」執行相同動作時使用，實作上要注意避免「無限迴圈」。

    從起始值開始進行條件判斷，若為 True 則執行程式，並在執行完成後將起始值加上步進值後再次判斷…執行，直到判斷為 False，迴圈即結束。

    for(起始值; 條件; 步進值){
	執行的程式
}

    例如：在網頁上產生 0-n 的文字

    for(var i = 0; i < 10; i++) {
	document.write(i);
}

    i++ 等同於 i = i + 1，撰寫時很常用前者；i-- 即意同 i = i - 1

    如果實作上迴圈是針對「次數、個數」，起始值通常會以 0 開始，步進值會以 1 為單位進行遞增或遞減，但也可以用其他數值。

    例如：0-100 間的偶數相加可以有以下兩種寫法

    //寫法一
var sum = 0;
for(var i = 0; i <= 100; i = i + 2) {
	sum = sum + i;
}
console.log(sum);

//寫法二
var sum = 0;
for(var i = 0; i <= 100; i++) {
	if(i % 2 == 0) {
		sum = sum + i;
	}
}
console.log(sum);

    可以練習 1-100 的連乘總合，(1 * 2 * 3 * ... * 100 = ?)

    練習時很常用迴圈搭配條件判斷繪製圖形，例：

    for(var row = 0; row < 10; row++) {
	for(var col = 0; col < 10; col++) {
		if(row % 2 == 0) {
			if(col % 2 == 0) {
				document.write('■');
			}else{
				document.write('□');
			}
		}else{
			if(col % 2 == 0) {
				document.write('□');
			}else{
				document.write('■');
			}
		}
	}
	document.write('<br>');
}
    # 實作練習：使用迴圈產生 100 個 <div> 加到網頁上
    var body = document.querySelector('body');
var num = 100;
for(var i = 0; i < 100; i++) {
	var div = document.createElement('div');
	div.innerHTML = '第' + i + '個';
	body.append(div);
}
    # 實作練習：使用迴圈產生 10 個 <div> 加到網頁上，點擊後目標 <div> 消失
    var body = document.querySelector('body');
var num = 10;
for(var i = 0; i < num; i++) {
	// document.createElement('div') 意即產生 <div></div>
	var div = document.createElement('div');

	// div.innerHTML 是指 <div>這裡的 HTML 內容</div>
	// div.innerHTML = '第' + i + '個' 意即將剛剛新產生的要素內容修改/取代。
	// 當 i = 0 的時候得到 <div>第 0 個</div>
	div.innerHTML = '第' + i + '個';

	// 綁定一個函式到 onclick 事件屬性上 (當被按下時執行函式)
	div.onclick = function() {
		this.style.display = 'none';
	}

	// body.append(div)` 是指將剛剛產生的 `<div>第 0 個</div>`加入 `<body>` 裡
	// 當 i = 0 的時候得到 `<body><div>第 0 個</div></body>`。
	// 隨著迴圈執行最後得到 `<body><div>第 0 個</div><div>第 1 個</div>...<div>第 9 個</div></body>`
	body.append(div);
}
